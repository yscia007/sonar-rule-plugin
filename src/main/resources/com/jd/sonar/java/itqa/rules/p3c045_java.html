所有的类都必须添加创建者和创建日期。

错误示例

  
 import com.jd.java.myClass;

public class ClassMustHaveAuthorRuleCheck {       // Noncompliant

    int     foo1() { return 0; }
    synchronized void    foo2(int value) { }
    int     foo3(int value) { return 0; }
    synchronized Object  foo4(int value) { return null; }
    private static boolean isUsingLock = false;
    public synchronized void operate() {
        isUsingLock = true;
        // do somthing
        isUsingLock = false;
    }
    private static String name = "";
    public void operate2() {
        synchronized (name)
        {
            if (name.equals(""))  {
                name = "boonya";
            }
        }
    }
    int     foo6(int value, String name) { return 0; }
    int     foo7(int ... values) { return 0;}
}

正确示例



import com.jd.java.myClass;

  /**
 * @author: zhangsan
 *  @date: 2019/2/21
 * */

public class ClassMustHaveAuthorRuleCheck {

    int     foo1() { return 0; }
    synchronized void    foo2(int value) { }
    int     foo3(int value) { return 0; }
    synchronized Object  foo4(int value) { return null; }
    private static boolean isUsingLock = false;
    public synchronized void operate() {
        isUsingLock = true;
        // do somthing
        isUsingLock = false;
    }
    private static String name = "";
    public void operate2() {
        synchronized (name)
        {
            if (name.equals(""))  {
                name = "boonya";
            }
        }
    }
    int     foo6(int value, String name) { return 0; }
    int     foo7(int ... values) { return 0;}
}
